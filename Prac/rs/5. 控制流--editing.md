# if

```rust
let number = 7;
if number < 5 {
    println!("小于 5");
} else {
    println!("大于等于 5");
}
```

# loop
```rust
let mut counter = 0;
loop {
  counter += 1;
  if counter >= 10 {
    break;
  }
}
```

# while
```rust
let mut number = 3;
while number != 0 {
    println!("{}", number);
    number -= 1;
}
```

# for

```rust
for number in 1..4 {
    println!("{}", number); // 包含1不包含4
}
for number in 1..=4 {
    println!("{}", number); // 包含1包含4
}
```

# match

```rust
enum Coin {
    Penny,
    Nickel,
    Dime,
    Quarter,
}

fn value_in_cents(mycoin: Coin) -> u8 {
    match mycoin {
        Coin::Penny => 1,
        Coin::Nickel => 5,
        Coin::Dime => 10,
        Coin::Quarter => 25,
    }
}
```
